cargaproducto js:
// Obtener el ID del producto de la URL
const params = new URLSearchParams(window.location.search);
const id = parseInt(params.get("id"));

// Buscar el producto
const producto = productos.find(p => p.id === id);

if (producto) {
  // Cargar informaci칩n del producto
  document.getElementById("nombreProducto").textContent = producto.nombre;
  document.getElementById("precioProducto").textContent = producto.precio;
  document.getElementById("descripcionProducto").textContent = producto.descripcion;

  // Configurar carrusel
  let indexImagen = 0;
  const imagenCarrusel = document.getElementById("imagenCarrusel");
  const prevBtn = document.getElementById("prev");
  const nextBtn = document.getElementById("next");

  // Funci칩n para mostrar imagen seg칰n el 칤ndice
  function mostrarImagen(i) {
    if (i < 0) indexImagen = producto.imagenes.length - 1;
    else if (i >= producto.imagenes.length) indexImagen = 0;
    else indexImagen = i;

      // Efecto fade-out y fade-in
  imagenCarrusel.style.opacity = 0; // desaparece
  setTimeout(() => {
    imagenCarrusel.src = producto.imagenes[indexImagen]; // cambia la imagen
    imagenCarrusel.style.opacity = 1; // aparece
  }, 200); // tiempo del fade-out
  }

  // Inicializar con la primera imagen
  mostrarImagen(0);

  // Eventos de los botones
  prevBtn.addEventListener("click", () => {
    indexImagen--; 
    mostrarImagen(indexImagen);
  });

  nextBtn.addEventListener("click", () => {
    indexImagen++; 
    mostrarImagen(indexImagen);
  });

  // Carrusel autom치tico cada 4 segundos
  setInterval(() => {
    indexImagen++;
    mostrarImagen(indexImagen);
  }, 6000);

} else {
  // Producto no encontrado
  document.querySelector(".producto-detalle").innerHTML = `
    <p>Producto no encontrado</p>
    <a href="../index.html">Volver a la tienda</a>
  `;
}
